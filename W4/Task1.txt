binary_interval (arr, length, minInterval, maxInterval)
	INTEGER left <- 0							// O(1)
	INTEGER right <- length - 1						// O(1)
	WHILE left ≤ right							// O(logn)
		INTEGER mid <- (left + right) / 2				// O(logn)
		IF arr of mid ≥ minInterval AND arr of mid ≤ maxInterval	// O(logn)
			RETURN 1						// O(1)
		ELSE								// O(logn)
			IF arr of mid < minInterval				// O(logn)
				left <- mid + 1					// O(logn)
			ELSE							// O(logn)
				IF arr of mid > maxInterval			// O(logn)
					right <- mid - 1		  	// O(logn)
	RETURN 0								// O(1)


Total runtime of the function: 4 * O(1) + 9 * O(logn)
